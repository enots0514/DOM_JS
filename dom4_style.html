바꾸기

       
       
 input 값에 따라 박스의 보더 색상을 바꾸기 

        btn.onclick = function(){
          잘못된 방법
            // box.style.border-color = colorInput.value;
              이 방법은 사용할 수 없다. 
          
          가능한 방법 1. - 속성자 선택자 활용방법
             box.style["border-color"] = colorInput.value; 
         
          가능한 방법 2. - 자바스크립트 명명자(카멜법) 활용방법
            box.style.borderColor = colorInput.value;
        }  
    
 자바스크립트에서는 class (예약어) 대신 className 을 사용한다.

완성 코드 확인하기

        input type="color" class="color-input" style = "width:70px;
        border-radius:10%; padding:0.3rem;" /   
        input type="button" class="btn" value="버튼"/  
     
        div id="box" style="width:100px; height:100px; 
        background-color:blue; border:5px solid red;
        margin-top:1rem; padding:1rem; text-align:center;"
         p 보더 색상 바꾸기 /p
         
         script
        let box = document.querySelector("#box");
        let btn = document.querySelector(".btn");
        let colorInput = document.querySelector(".color-input");

         btn.onclick = function(){
            // box.style["border-color"] = colorInput.value; 
           box.style.borderColor = colorInput.value;
         }
          /script
